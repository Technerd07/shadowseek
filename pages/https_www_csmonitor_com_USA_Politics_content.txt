	function submitForm( form )
	{
		let newsletterEmailField = form.querySelector('input[type="email"]');

		if (newsletterEmailField !== undefined )
		{
			let newsletterEmail = newsletterEmailField.value,
				rootURL = '/monitornewsletters?email=',
				newsletterSections = getCheckedNewsletters(form),
				trackingParam = getParam(form);

			if ( newsletterEmail !== undefined )
			{
				let signUpURL = rootURL + encodeURIComponent(newsletterEmail) + newsletterSections + trackingParam;
				window.location = signUpURL;
			}
		}
	}

	function removeMultipleListings()
	{
		let listings = document.querySelectorAll('#newsletter-listing');
		if ( listings.length > 1 )
		{
			for(let x=1; x<listings.length; x++)
			{
				listings[x].remove();
			}
		}
	}

	function getCheckedNewsletters( form )
	{
		let newsletterSections = '';
		if( form.hasAttribute('data-selection') )
		{
			let dataSelection = form.getAttribute('data-selection');
			if(dataSelection.length > 0)
			{
				dataSelection.split(' ').forEach( newsletter =>
				{
					newsletterSections += `&${newsletter}=true`;
				})
			}
		}
		return newsletterSections;
	}

	function getParam( form )
	{
		let param = setDefaultParam( form.parentElement );
		let newsletterBannerParent = form.parentElement.parentElement;
		if ( newsletterBannerParent.classList.contains( 'param-added' ) )
		{
			param = '&icid=mkt:web:' + newsletterBannerParent.getAttribute('data-param');
		}
		return param;
	}

	function setDefaultParam( formContainer )
	{
		let defaultParam = '&icid=mkt:web:banner',
			pageBody = document.querySelector('body');

		if ( formContainer )
		{
			if ( formContainer.classList.contains( 'books' ) )
			{
				defaultParam = '&icid=mkt:web:books';
			}
			else if ( formContainer.classList.contains( 'politics' ) )
			{
				defaultParam = '&icid=mkt:web:politics';
			}
			else if ( formContainer.classList.contains( 'culture-learning' ) )
			{
				defaultParam = '&icid=mkt:web:culture-learning';
			}
			else if ( formContainer.classList.contains( 'science' ) )
			{
				defaultParam = '&icid=mkt:web:science-nature';
			}
			else if ( formContainer.classList.contains( 'csperspective' ) )
			{
				defaultParam = '&icid=mkt:web:csp';
			}
			else if ( formContainer.classList.contains( 'commentary' ) )
			{
				defaultParam = '&icid=mkt:web:commentary';
			}

				defaultParam = '&icid=mkt:web:highlights';
			}
		}

		{
			defaultParam = '&icid=mkt:web:frontpage';
		}
		return defaultParam;
	}
